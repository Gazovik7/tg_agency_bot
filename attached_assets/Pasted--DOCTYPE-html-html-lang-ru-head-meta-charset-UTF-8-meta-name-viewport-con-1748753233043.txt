        function initHeatmap() {
            const heatmapBody = document.getElementById('heatmapBody');
            const hours = ['09:00', '10:00', '11:00', '12:00', '13:00', '14:00', '15:00', '16:00', '17:00', '18:00'];
            
            // –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–ª—É—á–∞–π–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ –¥–ª—è —Ç–µ–ø–ª–æ–≤–æ–π –∫–∞—Ä—Ç—ã
            const heatmapData = [];
            for (let hour = 0; hour < hours.length; hour++) {
                const row = [];
                for (let day = 0; day < 7; day++) {
                    row.push(Math.floor(Math.random() * 50) + 1);
                }
                heatmapData.push(row);
            }

            // –ù–∞—Ö–æ–¥–∏–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏–∏
            const maxValue = Math.max(...heatmapData.flat());

            hours.forEach((hour, hourIndex) => {
                const row = document.createElement('tr');
                row.innerHTML = `<td class="p-2 text-sm font-medium text-gray-600">${hour}</td>`;
                
                for (let day = 0; day < 7; day++) {
                    const value = heatmapData[hourIndex][day];
                    const intensity = value / maxValue;
                    const opacity = Math.max(0.1, intensity);
                    
                    const cell = document.createElement('td');
                    cell.className = 'p-1';
                    cell.innerHTML = `
                        <div class="heatmap-cell w-8 h-8 rounded flex items-center justify-center text-xs font-medium cursor-pointer" 
                             style="background-color: rgba(59, 130, 246, ${opacity}); color: ${intensity > 0.5 ? 'white' : '#374151'}"
                             title="${value} —Å–æ–æ–±—â–µ–Ω–∏–π">
                            ${value}
                        </div>
                    `;
                    row.appendChild(cell);
                }
                
                heatmapBody.appendChild(row);
            });
        }
